<script>

/** ====== 可改的設定 ====== **/

const ROUTES = [

  { id:"A", name:"A線 水頭翟山線", hint:"懷舊風情 × 海岸軍事地景（示意）" },

  { id:"B", name:"B線 古寧頭戰場線", hint:"戰史遺蹟 × 戰地記憶（示意）" },

  { id:"C", name:"C線 獅山民俗村線", hint:"在地文化 × 民俗聚落（示意）" },

  { id:"D", name:"D線 榕園太湖線", hint:"洋樓聚落 × 太湖風景（示意）" },

  { id:"E", name:"E線 尋找風獅爺線", hint:"風獅爺守護 × 集章超適合（示意）" },

  { id:"F", name:"F線 尋城趣文化小旅行", hint:"金城街區 × 文化散策（示意）" },

  { id:"G", name:"G線 金門大橋烈嶼線", hint:"大橋 × 小金門探索（示意）" },

];

 

// 每條路線示範 5 個章（你可改成真實站點數）

const STAMPS_PER_ROUTE = 5;

 

// 任務題庫（示範：之後可換成每站專屬題）

const QUESTIONS = [

  { q:"台灣好行最適合用什麼方式玩？", a:["自駕衝點","搭車集章闖關","只待在飯店"], correct:1 },

  { q:"抵達站點掃碼的好處是？", a:["增加等待時間","讓旅程更有互動","降低體驗"], correct:1 },

  { q:"集滿徽章後最適合做什麼？", a:["分享打卡導流","立刻刪除","不告訴任何人"], correct:0 },

];

 

/** ====== 儲存/讀取 ====== **/

const key = "km_taiwantrip_game_v1";

const state = JSON.parse(localStorage.getItem(key) || "{}");

 

function save(){ localStorage.setItem(key, JSON.stringify(state)); }

 

/** ====== UI ====== **/

const $ = (s)=>document.querySelector(s);

const routeList = $("#routeList");

const step1 = $("#step1");

const step2 = $("#step2");

const finish = $("#finish");

const currentRoutePill = $("#currentRoutePill");

const routeHint = $("#routeHint");

const progressText = $("#progressText");

const bar = $("#bar");

const stampsEl = $("#stamps");

const missionBox = $("#missionBox");

const missionActions = $("#missionActions");

const codeInput = $("#codeInput");

 

function initRouteList(){

  routeList.innerHTML = "";

  ROUTES.forEach(r=>{

    const d = document.createElement("div");

    d.className = "route";

    d.innerHTML = `<strong>${r.name}</strong><small>${r.hint}</small>`;

    d.onclick = ()=>selectRoute(r.id);

    routeList.appendChild(d);

  });

}

 

function ensureRouteState(routeId){

  if(!state.routes) state.routes = {};

  if(!state.routes[routeId]) state.routes[routeId] = { stamps:[], completed:false };

  save();

}

 

function selectRoute(routeId){

  state.currentRoute = routeId;

  ensureRouteState(routeId);

  save();

  render();

}

 

function render(){

  const r = state.currentRoute;

  if(!r){

    step1.style.display = "";

    step2.style.display = "none";

    finish.style.display = "none";

    return;

  }

  step1.style.display = "none";

  step2.style.display = "";

  const route = ROUTES.find(x=>x.id===r);

  currentRoutePill.textContent = `📍 目前路線：${route.id}（${route.name}）`;

  routeHint.textContent = route.hint;

 

  const stamps = state.routes[r].stamps || [];

  progressText.textContent = `${stamps.length}/${STAMPS_PER_ROUTE}`;

  bar.style.width = `${Math.min(100, (stamps.length/STAMPS_PER_ROUTE)*100)}%`;

 

  // render stamps

  stampsEl.innerHTML = "";

  for(let i=1;i<=STAMPS_PER_ROUTE;i++){

    const code = `${r}-${String(i).padStart(2,"0")}`;

    const done = stamps.includes(code);

    const s = document.createElement("div");

    s.className = "stamp" + (done ? " done":"");

    s.innerHTML = `<div style="font-weight:800">${done ? "✅ 已集章":"⬜ 未集章"}</div>

                   <div class="tiny muted">站點碼：${code}</div>`;

    stampsEl.appendChild(s);

  }

 

  if(stamps.length>=STAMPS_PER_ROUTE){

    finish.style.display = "";

    $("#rewardText").textContent = `你完成了 ${route.name} 的集章挑戰！可引導玩家去「合作店家」出示畫面換禮物，或領取抽獎碼。`;

    const coupon = makeCoupon(route.id);

    $("#couponHint").textContent = `示範抽獎碼：${coupon}（正式版可改成後端驗證/限次兌換）`;

    state.routes[r].completed = true;

    save();

  }else{

    finish.style.display = "none";

  }

}

 

function makeCoupon(routeId){

  // 只是示範：固定生成（不安全）。正式版要後端/簽章或一次性碼。

  const base = (state.deviceId ||= Math.random().toString(16).slice(2,10)).toUpperCase();

  save();

  return `KM-${routeId}-${base}`;

}

 

/** ====== 任務流程 ====== **/

function parseCode(raw){

  const t = (raw||"").trim().toUpperCase();

  // 支援 ?code=A-01 這種

  return t;

}

 

function nextQuestion(){

  const pick = QUESTIONS[Math.floor(Math.random()*QUESTIONS.length)];

  return pick;

}

 

function showMission(code){

  const r = state.currentRoute;

  const stamps = state.routes[r].stamps || [];

  missionActions.style.display = "none";

 

  // basic validate

  if(!/^[A-G]-\d{2}$/.test(code)){

    missionBox.className = "tiny muted";

    missionBox.textContent = "碼格式不正確。請輸入如 A-01、E-03 這類站點碼。";

    return;

  }

  if(code[0] !== r){

    missionBox.className = "tiny muted";

    missionBox.textContent = `你目前在 ${r} 線，但輸入的是 ${code}。請輸入同一路線的碼，或先換路線。`;

    return;

  }

  if(stamps.includes(code)){

    missionBox.className = "tiny muted";

    missionBox.textContent = `這個章（${code}）你已經拿過了！換下一站吧∼`;

    return;

  }

 

  const q = nextQuestion();

  missionBox.className = "tiny";

  missionBox.innerHTML = `<div style="font-weight:900;margin-bottom:8px">任務：回答問題即可集章（${code}）</div>

    <div style="margin-bottom:10px">${q.q}</div>`;

 

  missionActions.innerHTML = "";

  q.a.forEach((opt, idx)=>{

    const b = document.createElement("button");

    b.className = "btn secondary";

    b.textContent = `${idx+1}. ${opt}`;

    b.onclick = ()=>{

      if(idx === q.correct){

        // grant stamp

        state.routes[r].stamps.push(code);

        save();

        missionBox.innerHTML = `✅ 恭喜答對！你獲得一枚數位章：<strong>${code}</strong>`;

        missionActions.style.display = "none";

        render();

      }else{

        missionBox.innerHTML = `再試一次∼提示：答案跟「體驗更有互動/分享導流」有關。`;

      }

    };

    missionActions.appendChild(b);

  });

  missionActions.style.display = "flex";

}

 

/** ====== 事件 ====== **/

$("#changeRouteBtn").onclick = ()=>{

  state.currentRoute = null; save();

  codeInput.value = "";

  missionBox.textContent = "請先輸入/掃描站點碼。";

  missionActions.style.display="none";

  render();

};

 

$("#redeemBtn").onclick = ()=>{

  const code = parseCode(codeInput.value);

  showMission(code);

};

 

$("#demoBtn").onclick = ()=>{

  const r = state.currentRoute;

  const stamps = state.routes[r].stamps || [];

  // 找一個還沒拿過的碼

  let i=1;

  while(i<=STAMPS_PER_ROUTE){

    const code = `${r}-${String(i).padStart(2,"0")}`;

    if(!stamps.includes(code)){ codeInput.value = code; break; }

    i++;

  }

  showMission(parseCode(codeInput.value));

};

 

$("#shareBtn").onclick = async ()=>{

  const r = state.currentRoute;

  const route = ROUTES.find(x=>x.id===r);

  const text = `我剛完成「台灣好行金門線」${route.name} 集章任務！🚌✨ #台灣好行 #金門`;

  try{

    if(navigator.share){

      await navigator.share({ title:"好行金門集章", text });

    }else{

      await navigator.clipboard.writeText(text);

      alert("已複製分享文案到剪貼簿！");

    }

  }catch(e){

    alert("分享取消或不可用。");

  }

};

 

$("#keepPlayBtn").onclick = ()=>{

  state.currentRoute = null; save();

  render();

};

 

$("#resetBtn").onclick = ()=>{

  localStorage.removeItem(key);

  location.reload();

};

 

// 支援掃 QR 直接帶入：index.html?code=A-01

(function boot(){

  initRouteList();

  const url = new URL(location.href);

  const code = url.searchParams.get("code");

  if(code){

    // 若未選路線，先依 code 自動選

    const rid = (code||"").toUpperCase().trim()[0];

    if("ABCDEFG".includes(rid)) selectRoute(rid);

    codeInput.value = code.toUpperCase().trim();

    showMission(parseCode(codeInput.value));

  }else{

    render();

  }

})();

</script>
